继承

jicheng

  • 原型链继承（给原型对象添加的成员，实例对象可以继承到）
  1. 直接给原型对象添加 Person.prototype.xx = yyy;
    缺点：添加过多成员，以上写法重复，麻烦
function Person() {}
Person.prototype.color = "lime";
Person.prototype.sayHi = function () {
    console.log("hi");
}

var p = new Person();
console.log(p.color); // lime
p.sayHi();
  1. 原型替换 Person.prototype = {//...}
    缺点：造成了constructor属性丢失了
    解决：手动给新的原型对象补上
function Person() {}
Person.prototype = {
    constructor: Person,
    sayHi: function () {
        console.log("hi");
    },
    show: function () {
        console.log("show is ok");
    },
    // ...
};
var p = new Person();
p.sayHi();
p.show();
  1. Student.prototype = new Person();
    function Person () {
      this.head = '脑袋瓜子';
      this.emotion = ['喜', '怒', '哀', '乐']; //人都有喜怒哀乐
    }
    function Student(studentID) {
      this.studentID = studentID;
    }
    Student.prototype = new Person();

    var stu1 = new Student(1001);
    console.log(stu1.emotion); //['喜', '怒', '哀', '乐']

    stu1.emotion.push('愁');
    console.log(stu1.emotion); //["喜", "怒", "哀", "乐", "愁"]
    
    var stu2 = new Student(1002);
    console.log(stu2.emotion); //["喜", "怒", "哀", "乐", "愁"]

原型上任何类型的属性值都不会通过实例被重写，但是引用类型的属性值会受到实例的影响而修改

  • 借用构造函数继承

      function Person () {
    this.head = '脑袋瓜子';
    this.emotion = ['喜', '怒', '哀', '乐']; //人都有喜怒哀乐
  }
  function Student(studentID) {
    this.studentID = studentID;
    Person.call(this);
  }

  var stu1 = new Student(1001);
  console.log(stu1.emotion); //['喜', '怒', '哀', '乐']

  stu1.emotion.push('愁');
  console.log(stu1.emotion); //["喜", "怒", "哀", "乐", "愁"]
  
  var stu2 = new Student(1002);
  console.log(stu2.emotion); //["喜", "怒", "哀", "乐"]

    每个实例都拷贝一份，占用内存大，尤其是方法过多的时候

  • 组合继承
    既可以保证每个实例都有它自己的属性，又能做到对一些属性和方法的复用

    function Person () {
      this.head = '脑袋瓜子';
      this.emotion = ['喜', '怒', '哀', '乐']; //人都有喜怒哀乐
    }
    Person.prototype.eat = function () {
      console.log('吃吃喝喝');
    }
    Person.prototype.sleep = function () {
      console.log('睡觉');
    }
    Person.prototype.run = function () {
      console.log('快跑');
    }
    function Student(studentID) {
      this.studentID = studentID;
      Person.call(this);
    }
    
    Student.prototype = new Person();  
    Student.prototype.constructor = Student;  

    var stu1 = new Student(1001);
    console.log(stu1.emotion); //['喜', '怒', '哀', '乐']

    stu1.emotion.push('愁');
    console.log(stu1.emotion); //["喜", "怒", "哀", "乐", "愁"]
    
    var stu2 = new Student(1002);
    console.log(stu2.emotion); //["喜", "怒", "哀", "乐"]

    stu1.eat(); //吃吃喝喝
    stu2.run(); //快跑
    console.log(stu1.constructor);  //Student
